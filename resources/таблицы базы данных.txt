Table users {
  id integer [primary key]
  email varchar (128)
  login varchar (128)
  name varchar (128)
  birthday date
}

Table films {
  id integer [primary key]
  name varchar (255)
  description varchar
  release_date date
  duration bigint
  rating_mpa_id integer [ref: < film_ratings_mpa.id]
}

Table film_ratings_mpa {
  id integer [primary key]
  name varchar(16)
}

Table genre {
  id integer [primary key]
  name varchar (128)
}

Table film_genres {
  film_id integer [primary key]
  genre_id integer [primary key]
}

Table film_likes {
  film_id integer [primary key]
  user_id integer [primary key]
}

Table friends {
user_id integer [ref: > users.id]
friend_id integer [ref: > users.id]
friendship_status_id integer [ref: > friendship_status.id]
}

Table friendship_status {
  id integer [primary key]
  name varchar(255)
}

Ref: genre.id < film_genres.genre_id
